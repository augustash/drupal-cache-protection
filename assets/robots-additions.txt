
# Block AJAX endpoint from crawling. Drupal core's Views AJAX controller is an
# internal callback; polite crawlers that index it only fragment the page cache
# with no SEO value. The second rule covers language-prefixed paths
# (/en/views/ajax, /de/views/ajax, etc.).
Disallow: /views/ajax
Disallow: /*/views/ajax

# Block search routes from crawling. Search results pages are thin/duplicate
# content that hurts SEO and creates a vector for bot abuse. Sites that want
# search indexed can override with an Allow rule.
Disallow: /search

# Block faceted search URLs from crawling. Each unique facet combination is a
# distinct URL that bots can multiply to fragment the cache.
Disallow: /*?*f[0]=
Disallow: /*?*f%5B0%5D=

# Throttle AhrefsSiteAudit. Observed in the wild driving a 5-hour 499/502
# cascade by hammering uncacheable form pages with a distributed crawler swarm.
# Crawl-delay is interpreted per-bot, so 60s × ~30 concurrent crawlers caps
# aggregate near 0.5 req/sec — enough headroom for Pantheon's PHP workers.
# Real-time AhrefsBot (search index crawler) is intentionally not throttled.
# To run a legitimate site audit, raise the delay or remove this block.
User-agent: AhrefsSiteAudit
Crawl-delay: 60

# Block IbouBot. Self-identifying scraper with no SEO value, observed
# hammering form/auth pages.
User-agent: IbouBot
Disallow: /

# Reset to wildcard so any subsequent appends (e.g. site-level robots-additions)
# inherit User-agent: * cleanly instead of attaching to the last UA above.
User-agent: *
